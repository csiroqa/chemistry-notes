\section{\S 4.1 氧化还原反应复习}\label{sect:氧化还原反应复习}

    化合价升降

    \(\Rightarrow\) 电子（偏移或得失）

    \subsection{基本规律}\label{subsec:氧化还原基本规律}

        \subsubsection{守恒律}\label{subsubsec:守恒律}

            得电子总数 \(=\) 失电子总数

            （化合价升降守恒）

        \subsubsection{强弱律}\label{subsubsec:强弱律}

            \[
            \underset{\text{强}}{\text{氧化剂}} + \underset{\text{强}}{\text{还原剂}}
            \xleq{} \underset{\text{弱}}{\text{还原产物}} + \underset{\text{弱}}{\text{氧化产物}}
            \]

            氧化性：氧化剂 \(>\) 氧化产物

            还原性：还原剂 \(>\) 还原产物

            \begin{xmp}[强弱律例]{\cemh{NaClO + HCl}}
                \cemh{NaClO + 2HCl \xleq{} NaCl + Cl2 ^ + H2O}

                % [IMAGE: 单线桥示意图，箭头自 HCl 指出标注「还原剂，酸性」]
            \end{xmp}

        \subsubsection{转化律}\label{subsubsec:转化律}

            同种元素，化合价不交叉（归中）

            \textcolor{red}{只靠拢，不交叉。}

        \subsubsection{先后律}\label{subsubsec:先后律}

            相同条件下，还原性或氧化性更强的先反应。

            还原性：\cemh{I-} \(>\) \cemh{Fe^{2+}} \(>\) \cemh{Br-}

            氧化性：\cemh{I2} \(<\) \cemh{Fe^{3+}} \(<\) \cemh{Br2}

            \begin{xmp}[先后律例]{\cemh{Fe^{3+}} 与 \cemh{I-} 反应}
                \cemh{2Fe^{3+} + 2I- \xleq{} 2Fe^{2+} + I2}\qquad \(K = \num{6.2E7}\)

                \(c\uparrow\)，氧化性\(\uparrow\)（还原性同理）

                \textcolor{red}{浓度影响氧化还原性。}

                物质的氧化性（还原性）强弱，是\textcolor{red}{等浓度等条件}下的比较。如果氧化性（还原性）差异不大，浓度可能会改变二者的氧化性或还原性强弱关系。
            \end{xmp}

    \subsection{影响氧化还原反应的因素}\label{subsec:影响氧化还原反应的因素}

        \textcolor{red}{物质的氧化性、还原性会受浓度、\cemh{pH}、温度的影响。}

        \begin{xmp}[H+ 影响]{\cemh{H+} 影响氧化性}
            \cemh{MnO2 + 4HCl_{(conc)} \xleq{$\upDelta$} MnCl2 + Cl2 ^ + 2H2O}

            半反应：\cemh{MnO2 + 2e- + 4H+ \xleq{} Mn^{2+} + 2H2O}

            \(c(\cemh{H+})\uparrow\)，\cemh{MnO2} 氧化性\(\uparrow\)。

            \textcolor{red}{\cemh{H+} 影响氧化性。}
        \end{xmp}

        \begin{xmp}[王水]{王水溶金}
            王水（浓盐酸和浓硝酸）溶金时会有 \cemh{NO} 气体逸出，溶液中金元素主要以 \cemh{[AuCl4]-} 形式存在。

            浓盐酸的作用：\textcolor{red}{提供 \cemh{Cl-}，降低 \cemh{Au} 的还原性（形成配合物）；增大 \cemh{Cl-} 浓度。}

            把浓盐酸换成 \cemh{NaCl} 溶液后，还能溶金吗？\textcolor{red}{不能，需要 \cemh{H+} 的氧化性环境。}
        \end{xmp}

    \subsection{配平}\label{subsec:配平}

        \begin{enumerate}[label=\Roman*]
            \item \textcolor{red}{标出变化元素化合价}
            \item \textcolor{red}{标出化合价升降数目}（下角标计入升降数目中）
            \item \textcolor{red}{求最小公倍数定系数}
            \item \textcolor{red}{观察法确定其他系数}
            \item \textcolor{red}{用未使用的原子检查}
        \end{enumerate}

        \subsubsection{简单氧化还原}\label{subsubsec:配平-简单00}

            \begin{xmp}[配平例 1]{高锰酸钾与硫酸亚铁在酸性介质}
                \cemh{2KMnO4 + 10FeSO4 + 8H2SO4 \xleq{} K2SO4 + 2MnSO4 + 5Fe2(SO4)3 + 8H2O}
            \end{xmp}

            \begin{xmp}[配平例 2]{重铬酸钾氧化二氧化硫}
                \cemh{K2Cr2O7 + 3SO2 + H2SO4 \xleq{} Cr2(SO4)3 + K2SO4 + H2O}
            \end{xmp}

        \subsubsection{部分化合价变化}\label{subsubsec:配平-部分00}

            \begin{xmp}[配平例 3]{高锰酸钾氧化浓盐酸}
                \cemh{2KMnO4 + 16HCl \xleq{} 2MnCl2 + 5Cl2 ^ + 2KCl + 8H2O}

                \textcolor{blue}{先确定化合价全变的物质系数。\cemh{HCl} 部分被氧化，部分显酸性。}
            \end{xmp}

            \begin{xmp}[配平例 4]{镁与稀硝酸}
                \cemh{4Mg + 10HNO3 \xleq{} 4Mg(NO3)2 + NH4NO3 + 3H2O}
            \end{xmp}

            \begin{xmp}[配平例 5]{高锰酸钾与盐酸（另一种配比）}
                \cemh{KMnO4 + 5HCl + 3HCl \xleq{} MnCl2 + 5/2 Cl2 ^ + KCl + 4H2O}
            \end{xmp}

        \subsubsection{歧化与分解}\label{subsubsec:配平-歧化00}

            \begin{xmp}[配平例 6]{氯气在热碱中的歧化}
                \cemh{3Cl2 + 6NaOH \xleq{} 5NaCl + NaClO3 + 3H2O}
            \end{xmp}

            \begin{xmp}[配平例 7]{硫酸铵受热分解}
                \cemh{3(NH4)2SO4 \xleq{} N2 ^ + 4NH3 ^ + 3SO2 ^ + 6H2O}
            \end{xmp}

        \subsubsection{归中反应}\label{subsubsec:配平-归中00}

            \begin{xmp}[配平例 8]{一氧化氮与二氧化氮的归中}
                \cemh{NO + NO2 + 2NaOH \xleq{} 2NaNO2 + H2O}
            \end{xmp}

            \begin{xmp}[配平例 9]{氯酸钾与盐酸的归中}
                \cemh{2KClO3 + 4HCl \xleq{} 2ClO2 ^ + Cl2 ^ + 2KCl + 2H2O}
            \end{xmp}

        \subsubsection{多元素变化}\label{subsubsec:配平-多元素00}

            \begin{xmp}[配平例 10]{硫代硫酸钠被氯气氧化}
                \cemh{Na2S2O3 + 4Cl2 + 5H2O \xleq{} 2NaCl + 2H2SO4 + 8HCl}
            \end{xmp}

            \begin{xmp}[配平例 11]{四氧化三铁与氢碘酸}
                \cemh{Fe3O4 + 8HI \xleq{} I2 + 3FeI2 + 4H2O}
            \end{xmp}

            \begin{xmp}[配平例 12]{硫化亚铜与稀硝酸}
                \cemh{3Cu2S + 22HNO3 \xleq{} 6Cu(NO3)2 + 10NO ^ + 3H2SO4 + 8H2O}

                \textcolor{blue}{多组元素化合价变化，整体分析化合价变化直至满足一升一降。}
            \end{xmp}

            \begin{xmp}[配平例 13]{黑火药爆炸反应}
                \cemh{S + 2KNO3 + 3C \xleq{} K2S + N2 ^ + 3CO2 ^}

                \(\Rightarrow\) 转变比例，整体标价
            \end{xmp}

        \subsubsection{有机氧化还原}\label{subsubsec:配平-有机00}

            \begin{xmp}[配平例 14]{甲苯被高锰酸钾氧化}
                \cemh{5C6H5CH3 + 6KMnO4 + 9H2SO4 \xleq{} 5C6H5COOH + 3K2SO4 + 6MnSO4 + 14H2O}
            \end{xmp}

            \begin{xmp}[配平例 15]{乙醛被硼氢化钠还原}
                \cemh{CH3CHO + NaBH4 + H2O \xleq{} NaBO2 + CH3CH2OH}
            \end{xmp}

    \subsection{如何书写半反应式}\label{subsec:半反应书写}

        \begin{xmp}[半反应例 1]{氯化铁与铜}
            \cemh{2FeCl3 + Cu \xleq{} CuCl2 + 2FeCl2}

            氧化：\cemh{Cu - 2e- \xleq{} Cu^{2+}}

            还原：\cemh{Fe^{3+} + e- \xleq{} Fe^{2+}}
        \end{xmp}

        \begin{xmp}[半反应例 2]{硫化氢与氯化铁}
            \cemh{H2S + 2FeCl3 \xleq{} S v + 2FeCl2 + 2HCl}
            
            氧化：\cemh{H2S - 2e- \xleq{} S v + 2H+}

            还原：\cemh{Fe^{3+} + e- \xleq{} Fe^{2+}}
        \end{xmp}

        \begin{xmp}[半反应例 3]{铜与稀硝酸}
            \cemh{3Cu + 8HNO3 \xleq{} 3Cu(NO3)2 + 2NO ^ + 4H2O}
            
            氧化：\cemh{Cu - 2e- \xleq{} Cu^{2+}}

            还原：\cemh{4H+ + NO3- + 3e- \xleq{} NO ^ + 2H2O}
        \end{xmp}

        \textcolor{red}{书写步骤：}

        \begin{enumerate}
            \item 确定两极反应物产物
            \item 确定得失电子数相同
            \item 根据电性补 \cemh{OH-}、\cemh{H+}
            \item 书写半反应式
        \end{enumerate}

        \textcolor{blue}{配平时要注意反应式两边的氢、氧原子数目：}

        \begin{center}
        \small
        \begin{tabular}{|c|c|c|}
            \hline
            \textbf{条件} & \textbf{左边多氧} & \textbf{左边少氧} \\
            \hline
            酸性介质 & \cemh{+ 2H+ \xleq{} H2O} & \cemh{+ H2O \xleq{} 2H+} \\
            \hline
            碱性介质 & \cemh{+ H2O \xleq{} 2OH-} & \cemh{+ 2OH- \xleq{} H2O} \\
            \hline
        \end{tabular}
        \end{center}

        \textcolor{blue}{该方法半反应式、总反应式均适用。}
