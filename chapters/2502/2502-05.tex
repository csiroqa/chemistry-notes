\section*{专题——化合价与氧化还原配平}
\addcontentsline{toc}{section}{专题——化合价与氧化还原配平}
\label{sect:化合价和配平}

    \subsection{化合价的标定}\label{subsec:化合价标定}

        \begin{xmp}[标化合价]{标出下列红色元素的化合价}
            \cemh{CH4}、\cemh{NaH}、\cemh{Na2O}、\cemh{Na2O2}、\cemh{NH3}、\cemh{HNO3}、\cemh{NH4NO3}、\cemh{NaNO2}、\cemh{NO}、\cemh{NO2}

            \cemh{H3PO4}、\cemh{H2SO4}、\cemh{HClO4}、\cemh{FeS}、\cemh{Fe3O4}、\cemh{Na2FeO4}、\cemh{KMnO4}
        \end{xmp}

        \textcolor{blue}{氧化数（与化合价相似）：可以为分数；"平均"；有机物中 \cemh{H}、\cemh{O} 的氧化数为 \cemh{+1}、\cemh{-2}，计算 \cemh{C} 的氧化数。}

    \subsection{化合价升降法配平}\label{subsec:化合价升降法配平}

        \textbf{配平五步法}：

        \begin{enumerate}
            \item \textcolor{red}{标出变化元素化合价}
            \item \textcolor{red}{标出化合价升降数目}（下角标计入升降数目中）
            \item \textcolor{red}{求最小公倍数定系数}
            \item \textcolor{red}{观察法确定其他系数}
            \item \textcolor{red}{用未使用的原子检查}
        \end{enumerate}

        \subsubsection{简单氧化还原}\label{subsubsec:配平-简单}

            \begin{xmp}[配平例 1]{\cemh{KMnO4 + FeSO4 + H2SO4}}
                \cemh{KMnO4 + FeSO4 + H2SO4 \xleq{} K2SO4 + MnSO4 + Fe2(SO4)3 + H2O}

                \textcolor{blue}{系数：2、10、8、1、2、5、8}
            \end{xmp}

            \begin{xmp}[配平例 2]{\cemh{K2Cr2O7 + SO2 + H2SO4}}
                \cemh{K2Cr2O7 + SO2 + H2SO4 \xleq{} Cr2(SO4)3 + K2SO4 + H2O}

                \textcolor{blue}{系数：1、3、1、1、1、1}
            \end{xmp}

        \subsubsection{部分化合价变化}\label{subsubsec:配平-部分变化}

            \begin{xmp}[配平例 3]{\cemh{KMnO4 + HCl}}
                \cemh{KMnO4 + HCl \xleq{} MnCl2 + Cl2 ^ + KCl + H2O}

                \textcolor{blue}{先确定化合价全变的物质系数。\cemh{HCl} 部分被氧化，部分显酸性。}

                \textcolor{blue}{系数：2、16、2、5、2、8}
            \end{xmp}

            \begin{xmp}[配平例 4]{\cemh{Mg + HNO3}}
                \cemh{Mg + HNO3 \xleq{} Mg(NO3)2 + NH4NO3 + H2O}

                \textcolor{blue}{系数：4、10、4、1、3}
            \end{xmp}

        \subsubsection{歧化与分解}\label{subsubsec:配平-歧化分解}

            \begin{xmp}[配平例 5]{\cemh{Cl2 + NaOH}}
                \cemh{Cl2 + NaOH \xleq{} NaCl + NaClO3 + H2O}

                \textcolor{blue}{先确定生成物的系数。}

                \textcolor{blue}{系数：3、6、5、1、3}
            \end{xmp}

            \begin{xmp}[配平例 6]{\cemh{(NH4)2SO4} 分解}
                \cemh{(NH4)2SO4 \xleq{} N2 ^ + NH3 ^ + SO2 ^ + H2O}

                \textcolor{blue}{先确定生成物的系数。}

                \textcolor{blue}{系数：3、1、4、3、6}
            \end{xmp}

        \subsubsection{归中反应}\label{subsubsec:配平-归中}

            \begin{xmp}[配平例 7]{\cemh{NO + NO2 + NaOH}}
                \cemh{NO + NO2 + NaOH \xleq{} NaNO2 + H2O}

                \textcolor{blue}{先确定反应物的系数。}

                \textcolor{blue}{系数：1、1、2、2、1}
            \end{xmp}

            \begin{xmp}[配平例 8]{\cemh{KClO3 + HCl}}
                \cemh{KClO3 + HCl \xleq{} ClO2 ^ + Cl2 ^ + KCl + H2O}

                \textcolor{blue}{系数：2、4、2、1、2、2}
            \end{xmp}

        \subsubsection{多元素变化}\label{subsubsec:配平-多元素}

            \begin{xmp}[配平例 9]{\cemh{Cu2S + HNO3}}
                \cemh{Cu2S + HNO3 \xleq{} Cu(NO3)2 + NO ^ + H2SO4 + H2O}

                \textcolor{blue}{多组元素化合价变化，整体分析化合价变化直至满足一升一降。}

                \textcolor{blue}{系数：3、16、6、4、3、8}
            \end{xmp}

            \begin{xmp}[配平例 10]{\cemh{S + KNO3 + C}}
                \cemh{S + KNO3 + C \xleq{} K2S + N2 ^ + CO2 ^}

                \textcolor{blue}{锁定比例，整体标价。}

                \textcolor{blue}{系数：1、2、3、1、1、3}
            \end{xmp}

        \subsubsection{归零法}\label{subsubsec:配平-归零法}

            \begin{xmp}[配平例 11]{\cemh{Fe3C + HNO3}}
                \cemh{Fe3C + HNO3 \xleq{} Fe(NO3)3 + CO2 ^ + NO ^ + H2O}

                \textcolor{blue}{无法判断化合价，归零法——化合物的化合价代数和为零。}

                \textcolor{blue}{系数：1、22、3、1、9、11}
            \end{xmp}

        \subsubsection{离子方程式配平}\label{subsubsec:配平-离子方程}

            \textcolor{red}{离子方程式配平一般思路：电子守恒 \xleq{} 电荷守恒 \xleq{} 原子守恒}

            缺项一般不涉及化合价升降，根据水溶液的酸碱环境确定离子种类——酸性环境 \cemh{H+}，碱性环境 \cemh{OH-}。

            \begin{xmp}[配平例 12]{离子方程式配平}
                \cemh{AsO4^{3-} + I- + \_\_\_\_\_\_\_\_ \xleq{} AsO3^{3-} + I2 + \_\_\_\_\_\_\_\_}

                \textcolor{blue}{系数：1、2、4\cemh{H+}、1、1、2\cemh{H2O}}

                \cemh{Br2 + CO3^{2-} \xleq{} Br- + BrO3- + \_\_\_\_\_\_\_\_}

                \textcolor{blue}{系数：3、3、5、1、3\cemh{CO2}}
            \end{xmp}

    \subsection{综合例题}\label{subsec:配平-综合}

        \begin{xmp}[配平综合 1]{\cemh{NH4NO3} 分解}
            \cemh{\_\_\_ NH4NO3 \xleq{} \_\_\_ HNO3 + \_\_\_ N2 ^ + \_\_\_ H2O}

            \textcolor{blue}{先确定 \cemh{NH4NO3} 中 \cemh{N} 元素的平均化合价。\cemh{NH4+} 中 \cemh{N} 为 \cemh{-3}，\cemh{NO3-} 中 \cemh{N} 为 \cemh{+5}，\cemh{N2} 为 \cemh{0}。}

            \textcolor{blue}{系数：5、2、4、9}
        \end{xmp}

        \begin{xmp}[配平综合 2]{Mn(NO3)2 氧化}
            $\text{\_\_\_}$ \cemh{Mn(NO3)2 * 6H2O} $+$ $\text{\_\_\_}$ \cemh{(CH3CO)2O} \xleq{} \\
            \qquad $\text{\_\_\_}$ \cemh{(CH3COO)3Mn} $+$ $\text{\_\_\_}$ \cemh{HNO2} $+$ $\text{\_\_\_}$ \cemh{O2 ^} $+$ \cemh{40 CH3COOH}

            \textcolor{blue}{系数：8、40、8、16、1、40（提示：\cemh{Mn} 化合价不变，看 \cemh{N} 的变化：\cemh{NO3-} \xleq{} \cemh{NO2-}）}
        \end{xmp}
