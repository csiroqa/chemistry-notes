\section{\S 4.2 原电池}\label{sect:原电池}

    \subsection{Zn-Cu 原电池}\label{subsec:ZnCu原电池}

        % [IMAGE: Zn-Cu 原电池示意图，左电极 Zn（－），右电极 Cu（＋），外接电流表 A，电解质为稀硫酸]

        \cemh{Cu}：\textbf{正极}

        \cemh{Zn}：\textbf{负极}

        氧化：\cemh{Zn - 2e- \xleq{} Zn^{2+}}

        还原：\cemh{2H+ + 2e- \xleq{} H2 ^}

        \textcolor{red}{正正负负：}

        \begin{itemize}
            \item 正离子 \(\to\) 正极
            \item 负离子 \(\to\) 负极
        \end{itemize}

        电子：负极（\cemh{Zn}）\(\to\) 正极（\cemh{Cu}）（导线）

        盐桥：离子导体，平衡电荷

        \textcolor{blue}{\cemh{H+} 在正极得 \cemh{e-} 无阻更快}

        \textcolor{blue}{纯 \cemh{Zn}：理论无气泡}

        \textcolor{blue}{粗 \cemh{Zn}：速率快}

        \textcolor{red}{电极 = 电极材料 + 与电极接触的物质}

    \subsection{构成原电池的条件}\label{subsec:构成原电池条件}

        \begin{enumerate}
            \item 两个活泼性不同的能导电的电极

            \quad 活泼的作负极

            \quad 不活泼的作正极

            \item 电极间形成闭合回路（电子导体）
            \item 具有电解质溶液或熔融电解质（离子导体）
            \item 自发的氧化还原反应
        \end{enumerate}

        \cemh{Zn/Cu/ZnSO4} 溶液 \cmark

        正极：\cemh{O2 + 2H2O + 4e- \xleq{} 4OH-}

        \cemh{Fe/C/NaCl} 溶液 \cmark

        \textcolor{blue}{水中溶解的 \cemh{O2} 可以自发氧化大多数金属。}

        盐桥的作用：

        \begin{enumerate}
            \item 离子导体，平衡半电池电荷（离子库）
            \item 使 \cemh{Zn} 和 \cemh{Cu^{2+}} 不接触，电流稳定
        \end{enumerate}

        \textcolor{blue}{引入盐桥后内阻变大，双液原电池电流强度小。}

        \begin{center}
        \small
        \begin{tabular}{|c|c|c|}
            \hline
            \textbf{对比项} & \textbf{单液原电池} & \textbf{双液原电池} \\
            \hline
            装置 & 一个烧杯 & 两个烧杯（盐桥连接） \\
            \hline
            稳定性 & \cemh{Zn} 与 \cemh{Cu^{2+}} 直接接触，电流不稳 & \cemh{Zn} 与 \cemh{Cu^{2+}} 不接触，电流稳定 \\
            \hline
            电流强度 & 大 & 小（引入盐桥后内阻变大） \\
            \hline
        \end{tabular}

    \subsection{离子交换膜}\label{subsec:离子交换膜}

        \begin{enumerate}
            \item 正离子交换膜：只允许正离子
            \item 负离子交换膜：只允许负离子
            \item 质子交换膜：只允许 \cemh{H+}
        \end{enumerate}

        基于 \cemh{Zn + Cu^{2+} \xleq{} Cu + Zn^{2+}} 设计双液原电池：

        氧化：\cemh{Zn - 2e- \xleq{} Zn^{2+}}

        还原：\cemh{Cu^{2+} + 2e- \xleq{} Cu}

    \subsection{浓差电池}\label{subsec:浓差电池}

        浓差电池：利用浓淡差。

        金属失电子难易与溶液浓度相关。

        \begin{xmp}[浓差电池]{不同浓度 \cemh{Ni^{2+}} 的浓差电池}
            两烧杯均为 \cemh{Ni} 电极，\cemh{Ni^{2+}} 浓度不同（如 \SI{1.00}{\mole\per\litre} 和 \SI{1.00E-3}{\mole\per\litre}）。

            \textcolor{blue}{高浓度区：\cemh{Ni^{2+}} 得电子被还原（正极）}

            \textcolor{blue}{低浓度区：\cemh{Ni} 失电子被氧化（负极）}

            \textcolor{red}{浓差电池——电解质浓度差异产生电势。}
        \end{xmp}

        \textcolor{blue}{电鳗：电细胞两侧离子浓度差异，产生电势约 \SI{0.1}{V}，数千个串联可达 \SI{500}{V}。}
