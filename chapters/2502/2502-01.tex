\section{\S 3.1 原电池}\label{sect:原电池}

    \subsection{原电池的工作原理}\label{subsec:原电池工作原理}

        \textcolor{red}{原电池：将化学能转化为电能的装置。}

        \subsubsection{Zn-Cu 原电池实验}\label{subsubsec:ZnCu原电池实验}

            \textbf{实验}：锌片、铜片同时插入稀硫酸中，用导线连接。

            \textbf{现象}：铜片有气泡，锌片溶解，电流表指针偏转。

            \textbf{电极反应}：

            \qquad 负极（\cemh{Zn}）：\cemh{Zn - 2e- \xleq{} Zn^{2+}}（氧化反应）

            \qquad 正极（\cemh{Cu}）：\cemh{2H+ + 2e- \xleq{} H2 ^}（还原反应）

            总反应：\cemh{Zn + 2H+ \xleq{} Zn^{2+} + H2 ^}

            \textcolor{red}{电子流向：Zn \xleq{} 导线 \xleq{} Cu}

            \textcolor{red}{电流方向：Cu \xleq{} 导线 \xleq{} Zn}

        \subsubsection{电势差的产生}\label{subsubsec:电势差产生}

            任何金属与水溶液界面上，同时存在：

            \qquad \cemh{M - $n$ e- \xleq{} M^{n+}} / \cemh{M^{n+} + $n$ e- \xleq{} M}

            金属活动性不同，失电子能力不同，因此在两电极间产生电势差。

            \textcolor{blue}{高电势能：负极；低电势能：正极。}

            \textcolor{blue}{注：电势越高，电势能越低。}

        \subsubsection{盐桥的作用}\label{subsubsec:盐桥作用}

            单液原电池中，\cemh{Zn} 和 \cemh{Cu^{2+}} 接触直接发生氧化还原反应，\cemh{Zn} 表面被 \cemh{Cu} 覆盖。

            \textcolor{red}{盐桥（如 \cemh{KNO3} 的琼脂溶液）的作用：}

            \begin{enumerate}
                \item \textcolor{red}{离子导体，平衡半电池电荷（离子库）}
                \item \textcolor{red}{使 \cemh{Zn} 和 \cemh{Cu^{2+}} 不接触，电流稳定}
            \end{enumerate}

            盐桥中 \cemh{K+} 向正极区（\cemh{CuSO4} 溶液）迁移，\cemh{NO3-} 向负极区（\cemh{ZnSO4} 溶液）迁移。

            \textcolor{blue}{无盐桥时，电荷积累阻碍了电子得失，反应停止。}

    \subsection{构成原电池的条件}\label{subsec:构成原电池条件}

        \begin{enumerate}
            \item \textcolor{red}{两个活动性不同的能导电的电极}——较活泼的金属作负极，较不活泼的金属（或石墨）作正极
            \item \textcolor{red}{电极间形成闭合回路}（电子导体）
            \item \textcolor{red}{具有电解质溶液或熔融电解质}（离子导体）
            \item \textcolor{red}{自发的氧化还原反应}
        \end{enumerate}

        \begin{xmp}[自发反应判断]{Fe/C/NaCl 体系}
            \textbf{电极}：\cemh{Fe / C / NaCl} 溶液

            正极：\cemh{O2 + 2H2O + 4e- \xleq{} 4OH-}

            \textcolor{blue}{水中溶解的 \cemh{O2} 可以自发氧化大多数金属。}
        \end{xmp}

    \subsection{单液原电池与双液原电池}\label{subsec:单液双液原电池}

        \begin{center}
        \small
        \begin{tabular}{|c|c|c|}
            \hline
            \textbf{对比项} & \textbf{单液原电池} & \textbf{双液原电池} \\
            \hline
            装置 & 一个烧杯 & 两个烧杯（盐桥连接） \\
            \hline
            稳定性 & \cemh{Zn} 与 \cemh{Cu^{2+}} 直接接触，电流不稳 & \cemh{Zn} 与 \cemh{Cu^{2+}} 不接触，电流稳定 \\
            \hline
            内阻 & 小 & 大（引入盐桥后内阻变大） \\
            \hline
            电流强度 & 大 & 小 \\
            \hline
        \end{tabular}
        \end{center}

        \subsubsection{离子交换膜}\label{subsubsec:离子交换膜}

            高中常见的三种离子交换膜：

            \begin{itemize}
                \item \textbf{阳离子交换膜}：只允许阳离子通过
                \item \textbf{阴离子交换膜}：只允许阴离子通过
                \item \textbf{质子交换膜}：只允许 \cemh{H+} 通过
            \end{itemize}

        \begin{xmp}[双液原电池设计]{设计 \cemh{Fe^{3+}/I-} 双液原电池}
            基于 \cemh{2Fe^{3+} + 2I- \xleq{} 2Fe^{2+} + I2} 设计双液原电池。

            负极（\cemh{C | I-}）：\cemh{2I- - 2e- \xleq{} I2}

            正极（\cemh{Fe^{3+} | C}）：\cemh{Fe^{3+} + e- \xleq{} Fe^{2+}}

            \textcolor{red}{电极 \textcolor{black}{=}} 电极材料 + 与材料接触的物质
        \end{xmp}

    \subsection{浓差电池}\label{subsec:浓差电池}

        金属失电子难易与溶液浓度相关。

        \begin{xmp}[浓差电池]{不同浓度 \cemh{Ni^{2+}} 的浓差电池}
            两烧杯均为 \cemh{Ni} 电极，\cemh{Ni^{2+}} 浓度不同\\*
            \qquad（如\SI{1.00}{\mole\per\litre}和\SI{1.00E-3}{\mole\per\litre}）。

            \textcolor{blue}{高浓度区：\cemh{Ni^{2+}} 得电子被还原（正极）}

            \textcolor{blue}{低浓度区：\cemh{Ni} 失电子被氧化（负极）}

            \textcolor{red}{浓差电池——电解质浓度差异产生电势。}
        \end{xmp}

        \textcolor{blue}{电鳗：电细胞两侧离子浓度差异，产生电势约\SI{0.1}{V}，数千个串联可达\SI{500}{V}。}
