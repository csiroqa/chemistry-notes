\section{\S 4.3 电解池}\label{sect:电解池}

    电解：化合物得到金属的最终手段

    阴极：与外电源负极相连

    阳极：与外电源正极相连

    电离 \(\Rightarrow\) 迁移 \(\Rightarrow\) 放电

    （异性相吸）

    正负离子在两极竞争放电。

    \subsection{电解 \texorpdfstring{\cemh{CuCl2}}{CuCl2} 溶液}\label{subsec:电解CuCl2}

        阴：\cemh{Cu^{2+} + 2e- \xleq{} Cu}

        \cemh{H+} 得 \cemh{e-}：\cemh{Cu^{2+}} \(>\) \cemh{H+}

        阳：\cemh{2Cl- - 2e- \xleq{} Cl2 ^}

        \cemh{OH-} 失 \cemh{e-}：\cemh{Cl-} \(>\) \cemh{OH-}

        总：\cemh{CuCl2 \xleq{通电} Cu + Cl2 ^}

    \subsection{离子的放电顺序}\label{subsec:离子放电顺序}

        \subsubsection{阴极放电顺序（阳离子得电子）}\label{subsubsec:阴极放电顺序}

            \textcolor{red}{\cemh{Ag+} \(>\) \cemh{Fe^{3+}} \(>\) \cemh{Cu^{2+}} \(>\) \cemh{H+}}\text{（酸）}\textcolor{red}{\(>\) \cemh{Fe^{2+}} \(>\) \cemh{Zn^{2+}} \(>\) \cemh{H+}}\text{（水）}\textcolor{red}{\(>\) \cemh{Al^{3+}} \(>\) \cemh{Mg^{2+}} \(>\) \cemh{Na+} \(>\) \cemh{Ca^{2+}} \(>\) \cemh{K+}}

            \textcolor{blue}{注：\cemh{H+} 氢前铝后看浓度。}

        \subsubsection{阳极放电顺序（阴离子失电子）}\label{subsubsec:阳极放电顺序}

            前提：惰性电极（\cemh{C}、\cemh{Pt}、\cemh{Au}）

            \textcolor{red}{\cemh{I-} \(>\) \cemh{Br-} \(>\) \cemh{Cl-} \(>\) \cemh{OH-} \(>\) （族价）含氧酸根 \(>\) \cemh{F-}}

            阳极：活性电极

            \textcolor{blue}{活性电极（除 \cemh{Pt}、\cemh{Au} 外的金属）优先于阴离子放电——阳极溶解。}

        \textcolor{red}{阳}极出\textcolor{red}{氧卤}，\textcolor{red}{阴}极\textcolor{red}{氢金属}

    \subsection{电极对电极反应的影响}\label{subsec:电极对电极反应的影响}

        \cemh{Cu + 2H+ \xleq{通电} Cu^{2+} + H2 ^}

        阴极：先产生气泡，后析出红色固体

        \textcolor{red}{电解使非自发进行的氧化还原反应得以发生。}

        \textcolor{blue}{若阴极把 \cemh{Cu} 换成石墨：阴极区先产生气泡，后析出红色固体（\cemh{Cu^{2+}} 与 \cemh{H+} 竞争的结果）。}

    \subsection{电解的应用}\label{subsec:电解的应用}

        \subsubsection{精炼铜}\label{subsubsec:精炼铜}

            % [IMAGE: 电解精炼铜示意图，标注：e⁻流入、e⁻流出、阳、阴、粗铜、精铜、阳极泥（Ag、Au、Pt等）、硫酸酸化的CuSO₄溶液]

            \cemh{Fe - 2e- \xleq{} Fe^{2+}}

            \cemh{Zn - 2e- \xleq{} Zn^{2+}}

            \cemh{Ni - 2e- \xleq{} Ni^{2+}}

            \cemh{Cu - 2e- \xleq{} Cu^{2+}}

            阳极溶解

            \cemh{Cu^{2+} + 2e- \xleq{} Cu}

            阴极沉积

            \textcolor{red}{溶少积多}

            \textcolor{blue}{\cemh{Cu^{2+}} 浓度略降（以 \SI{2}{\mole} \cemh{e-} 为例，阳极溶解少于阴极沉积）。}

        \subsubsection{电镀}\label{subsubsec:电镀}

            铜（镀件/阴极）上镀银（镀层/阳极）

            电镀液：\cemh{AgNO3}

            \textcolor{red}{电镀液的浓度不变，镀层更均匀。}

        \subsubsection{处理含 \texorpdfstring{\cemh{Cr2O7^{2-}}}{重铬酸根} 酸性废水}\label{subsubsec:废水处理}

            \cemh{Cr(OH)3 v}

            \cemh{6Fe^{2+} + Cr2O7^{2-} + 14H+ \xleq{} 6Fe^{3+} + 2Cr^{3+} + 7H2O}

            \cemh{pH}\(\uparrow\)，

            \cemh{Cr(OH)3 v + Fe(OH)3 v}

            \cemh{Fe} 为阳极：

            \cemh{Fe - 2e- \xleq{} Fe^{2+}}

            \cemh{Fe} 为阴极：

            \cemh{2H+ + 2e- \xleq{} H2 ^}

            \textcolor{blue}{\cemh{NaCl} 溶液增加导电性（\cemh{Cl-} 在阳极不放电，因为 \cemh{Fe} 是活性电极优先溶解）。}

        \subsubsection{氯碱工业}\label{subsubsec:氯碱工业}

            % [IMAGE: 氯碱工业电解装置示意图，标注：Cl₂↑、↑NaOH、NaCl↑、阳离子交换膜、↑H₂O]

            阳极：\cemh{2Cl- - 2e- \xleq{} Cl2 ^}

            阴极：\cemh{2H2O + 2e- \xleq{} 2OH- + H2 ^}

            总反应：\cemh{2NaCl + 2H2O \xleq{电解} 2NaOH + H2 ^ + Cl2 ^}

            \textcolor{red}{使用阳离子交换膜，防止 \cemh{Cl2} 与 \cemh{NaOH} 反应。}

    \subsection{电解过程中的 pH 变化}\label{subsec:电解pH变化}

        \textcolor{red}{极区 pH 变化看电极反应，不是离子迁移。}

        \textcolor{red}{极区 pH 变化 \(\neq\) 溶液 pH 变化。}
