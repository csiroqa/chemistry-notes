\chapter{晶体结构与性质} \label{chap:晶体结构与性质}

    \section{晶体的基本概念}\label{sect:晶体基本概念}

        \textbf{晶体}：具有规则几何外形和固定熔点的固体。

        \textbf{非晶体}：没有固定熔点，一般也没有规则几何外形的固体（如石蜡、玻璃、橡胶）。

        是否是晶体，与颗粒大小无关。

        \subsection{晶体的自范性}

            \textcolor{red}{在适宜的条件下，晶体能够自发地呈现封闭的、规则的几何多面体外形。}

            晶体的自范性是晶体中微粒在三维空间里呈现周期性有序排列的宏观表现。非晶体中微粒的排列则相对无序，因而无自范性。

        \subsection{晶体的各向异性}

            \textcolor{red}{晶体中不同的方向上具有不同的物理性质。}

            \begin{itemize}
                \item \textbf{导热性}：云母在不同方向上的导热性不同
                \item \textbf{导电性}：石墨在层平行方向上的电导率是层垂直方向上的 1 万倍
                \item \textbf{硬度}：方解石在不同方向上的刻痕深浅不同
                \item \textbf{折光率}：红宝石在不同方向上的颜色深浅不同
            \end{itemize}

            \textcolor{blue}{非晶体——各向同性。}

    \section{晶胞}\label{sect:晶胞}

        \textcolor{red}{晶胞：晶体结构中基本的重复单元。}

        常规的晶胞都是平行六面体。整块晶体可看作是数量巨大的晶胞"无隙并置"而成。

        \textbf{无隙并置}："无隙"指相邻晶胞之间无任何间隙；"并置"指所有晶胞都是平行排列的，取向相同。

        \textcolor{red}{晶胞的选取关键不是最小，而是无隙并置。}

        \textcolor{blue}{同一晶体内，所有晶胞的形状及其内部的原子种类、个数及几何排列（包括取向）是完全相同的。}

        \subsection{配位数}

            \textcolor{red}{配位数：距离某个原子最近且距离相等的原子个数。}

        \subsection{晶胞中粒子个数的计算}

            \textcolor{red}{均摊法}：晶胞外围的原子并不为该晶胞所独有，而是共用的。

            \begin{center}
            \begin{tabular}{|c|c|c|}
                \hline
                \textbf{位置} & \textbf{贡献份额} & \textbf{说明} \\
                \hline
                顶角 & \(1/8\) & 每个顶角原子被 8 个晶胞共用 \\
                \hline
                棱上 & \(1/4\) & 每个棱上原子被 4 个晶胞共用 \\
                \hline
                面上（面心） & \(1/2\) & 每个面心原子被 2 个晶胞共用 \\
                \hline
                内部（体心） & \(1\) & 完全属于该晶胞 \\
                \hline
            \end{tabular}
            \end{center}

            \textcolor{blue}{以上适用于平行六面体晶胞。}

    \section{四类典型晶体}\label{sect:四类典型晶体}

        \subsection{离子晶体}

            \textcolor{red}{正负离子通过离子键结合而成的晶体。}

            \begin{itemize}
                \item \textbf{基本结构微粒}：正、负离子
                \item \textbf{作用力}：离子键（既无方向性，也无饱和性）
                \item \textbf{物质类别}：离子化合物
                \item 只要空间条件允许，一个离子将尽可能多地吸引带相反电荷的离子排列在其周围
            \end{itemize}

            \textbf{离子晶体的性质特征}：

            \begin{enumerate}
                \item 熔沸点高且难挥发
                \item 硬度较大，不具有延展性
                \item 固态时不导电，熔融状态和溶于水后能导电
                \item 大多数易溶于极性溶剂，难溶于非极性溶剂
            \end{enumerate}

            \textcolor{red}{影响离子晶体熔点的因素}：\textcolor{blue}{①离子电荷\quad ②离子半径\quad（电荷 \(>\) 半径）}

            \textcolor{blue}{离子越大、含原子越多，熔点越低。}

            \textbf{离子液体}：由离子半径较大的正离子（或负离子）构成，结构不对称，较难在空间紧密堆积，正、负离子之间的静电作用小，熔点低。

            \begin{xmp}[离子晶体]{典型离子晶体晶胞}
                \textbf{NaCl 型}（配位数 6:6）：
                晶胞中含 \cemh{Na+} \(4\)个、\cemh{Cl-} \(4\)个。\\
                每个 \cemh{Na+} 周围最近邻 \cemh{Cl-} \(6\)个。
                每个 \cemh{Na+} 周围最近邻 \cemh{Na+} \(12\)个。

                \textbf{CsCl 型}（配位数 8:8）：
                晶胞中含 \cemh{Cs+} \(1\)个、\cemh{Cl-} \(1\)个。\\
                每个 \cemh{Cs+} 周围最近邻 \cemh{Cl-} \(8\)个。
                每个 \cemh{Cs+} 周围最近邻 \cemh{Cs+} \(6\)个。

                \textbf{CaF\(_2\)型}（配位数 8:4）：
                晶胞中含 \cemh{Ca^{2+}} \(4\)个、\cemh{F-} \(8\)个。\\
                每个 \cemh{Ca^{2+}} 周围最近邻 \cemh{F-} \(8\)个。
                每个 \cemh{F-} 周围最近邻 \cemh{Ca^{2+}} \(4\)个。

                \textbf{ZnS 型}（配位数 4:4）：
                晶胞中含 \cemh{Zn^{2+}} \(4\)个、\cemh{S^{2-}} \(4\)个。\\
                每个 \cemh{Zn^{2+}} 周围最近邻 \cemh{S^{2-}} \(4\)个。
            \end{xmp}

            \textcolor{blue}{离子化合物的化学式只表示离子晶体中正、负离子的最简个数比，而不是分子式。}

        \subsection{共价晶体}

            \textcolor{red}{相邻原子之间通过共价键而形成的三维空间网状结构的晶体。}

            \begin{itemize}
                \item \textbf{基本结构微粒}：原子
                \item \textbf{作用力}：共价键
                \item \textbf{常见物质}：金刚石、晶体硅（\cemh{Si}）、水晶（\cemh{SiO2}）、金刚砂（\cemh{SiC}）、单质锗（\cemh{Ge}）、氮化硼（\cemh{BN}）、氮化硅（\cemh{Si3N4}）等
            \end{itemize}

            \textbf{共价晶体的性质特征}：

            \begin{enumerate}
                \item 熔沸点很高，硬度很大
                \item 一般不导电，但晶体硅、锗是半导体
                \item 难溶于一般溶剂
            \end{enumerate}

            \textcolor{red}{影响共价晶体熔沸点和硬度的因素}：共价键的键能。结构相似的共价晶体：原子半径越小，键长越短，键能越大，晶体熔点越高，硬度越大。

            \begin{xmp}[共价晶体]{典型共价晶体结构}
                \textbf{金刚石}：
                每个 \cemh{C} 原子以 \cemh{sp^3} 杂化与另 4 个 \cemh{C} 原子以单键相连。
                键长 \SI{154}{\pm}，键角\ang{109;28}。
                \SI{12}{\g}金刚石中含 \cemh{C} 原子 \(N_\text{A}\)个，\cemh{C-C} 键 \(2N_\text{A}\)根。
                最小环由 6 个 \cemh{C} 原子组成，每个最小环实际占有 0.5 个 \cemh{C} 原子。
                每个 \cemh{C} 原子被 12 个最小环共用，每根 \cemh{C-C} 键被 6 个最小环共用。
                金刚石晶胞含 \cemh{C} 原子 \(8\)个，配位数 \(4\)。

                \textbf{SiO\(_2\)}：
                晶胞中含 \cemh{Si} \(8\)个、\cemh{O} \(16\)个。
                每个 \cemh{Si} 连接 4 个 \cemh{O}，每个 \cemh{O} 连接 2 个 \cemh{Si}。
                \SI{1}{\mole} \cemh{SiO2} 中含 \cemh{Si-O} 键 \(4N_\text{A}\)根。
                最小环是 12 元环（6 个 \cemh{Si} + 6 个 \cemh{O}）。
            \end{xmp}

            \textcolor{blue}{共价键有方向性，因此共价晶体中没有密堆积的特征（与离子晶体和金属晶体不同）。}

        \subsection{分子晶体}

            \textcolor{red}{分子通过分子间作用力（范德华力）形成的晶体。}

            \begin{itemize}
                \item \textbf{基本结构微粒}：分子（或稀有气体原子）
                \item \textbf{作用力}：分子间作用力（范德华力），分子内原子间为共价键
                \item \textbf{常见物质}：非金属单质（除金刚石、晶体硅、单质硼）、共价化合物（除共价晶体）、稀有气体
            \end{itemize}

            \textbf{分子晶体的性质特征}：

            \begin{enumerate}
                \item 熔沸点低，很多常温下是气体或液体，固体有些易升华（干冰、碘、萘等）
                \item 硬度小
                \item 不导电（固态、熔融状态均不导电），少数溶于水导电（酸）
                \item 溶解性一般符合相似相溶经验规则
            \end{enumerate}

            \textcolor{red}{影响分子晶体熔沸点的因素}（分子间作用力的大小）：
            \begin{enumerate}
                \item 有无分子间氢键
                \item 组成和结构相似时，相对分子质量的大小
                \item 相对分子质量相近时，分子极性的大小
            \end{enumerate}

            \textbf{分子密堆积}：大多数分子晶体中，以一个分子为中心，其周围最多可以有 12 个紧邻的分子（面心立方排列）。

            \begin{xmp}[分子晶体]{干冰与冰}
                \textbf{干冰}（\cemh{CO2}）：
                晶胞为面心立方，含 4 个 \cemh{CO2} 分子。
                分子密堆积。

                \textbf{冰}（\cemh{H2O}）：
                每个 \cemh{H2O} 分子依靠氢键与周围位于正四面体顶角的 4 个水分子结合。
                由于氢键的方向性，冰中水分子形成空隙较大的网状体——\textcolor{red}{非密堆积}。
                \SI{1}{\mole} \cemh{H2O} 形成的冰中含有氢键 \SI{2}{\mole}。
                冰融化后分子间空隙减小，故冰的密度比水小。
            \end{xmp}

        \subsection{金属晶体}

            \textcolor{red}{通过金属键形成的晶体。}

            \textbf{电子气理论}：金属正离子间依靠自由电子而产生的强烈相互作用，叫做金属键。本质是静电作用力，没有方向性和饱和性。

            \begin{itemize}
                \item \textbf{结构微粒}：金属正离子和自由电子
                \item \textbf{作用力}：金属键
                \item \textbf{存在}：金属单质或合金
            \end{itemize}

            \textbf{金属晶体的性质特征}：

            \begin{enumerate}
                \item 具有金属光泽，良好的延展性、导电性和导热性等
                \item 金属的熔点、硬度相差很大（\cemh{Hg} 熔点\SI{-39}{\celsius}，\cemh{W} 熔点\SI{3410}{\celsius}）
                \item 固态、液态均导电
            \end{enumerate}

            \textcolor{red}{用电子气理论解释金属通性}：

            \begin{itemize}
                \item \textbf{金属光泽}：自由电子吸收所有频率的光并很快放出
                \item \textbf{导电性}：自由电子在外加电场下定向运动形成电流
                \item \textbf{导热性}：自由电子运动时与金属离子碰撞传递能量
                \item \textbf{延展性}：金属键无方向性，各层滑动后仍保持相互作用
            \end{itemize}

            \textcolor{red}{影响金属晶体熔沸点的因素}（金属键的强弱）：
            \begin{enumerate}
                \item 金属正离子的电荷数
                \item 金属正离子的半径
                \item 金属晶体的堆积方式
            \end{enumerate}

            \begin{xmp}[金属晶体]{金属晶胞的四种堆积方式}
                \textbf{简单立方堆积}（非密堆积）：
                晶胞含原子 \(1\)个，配位数 \(6\)，空间利用率 \(52.36\%\)。
                代表金属：\cemh{Po}（钋）。

                \textbf{体心立方堆积}（密堆积）：
                晶胞含原子 \(2\)个，配位数 \(8\)，空间利用率 \(68.02\%\)。
                代表金属：\cemh{Na}、\cemh{K}、\cemh{Fe} 等。

                \textbf{面心立方堆积}（最密堆积）：
                晶胞含原子 \(4\)个，配位数 \(12\)，空间利用率 \(74.05\%\)。
                代表金属：\cemh{Cu}、\cemh{Ag}、\cemh{Au} 等。

                \textbf{六方堆积}（最密堆积）：
                晶胞含原子 \(6\)个，配位数 \(12\)，空间利用率 \(74.05\%\)。
                代表金属：\cemh{Mg}、\cemh{Zn}、\cemh{Ti}。
            \end{xmp}

    \section{混合型晶体与过渡晶体}\label{sect:混合型过渡晶体}

        \subsection{混合型晶体——石墨}

            \begin{itemize}
                \item \textbf{结构}：层状结构，每层为平面正六边形网状结构
                \item \textbf{层内}：每个 \cemh{C} 原子 \cemh{sp^2} 杂化，与另 3 个 \cemh{C} 原子相连（键长 \SI{142}{\pm}），还有垂直于碳环平面的未杂化 2p 轨道形成遍及整个平面的离域\(\uppi\)键
                \item \textbf{层间}：范德华力（容易滑动，质地很软——分子晶体特征）
                \item 离域\(\uppi\)键中的电子可在层内自由运动，因此石墨能导电——金属晶体特征
                \item 层内共价键断裂能量高，熔点很高（比金刚石高）——共价晶体特征
            \end{itemize}

            \textcolor{red}{石墨兼具三类晶体的性质特点，属于混合型晶体。}

            石墨中键长 \SI{142}{\pm}，金刚石中键长 \SI{154}{\pm}，因此石墨键能更大、熔点更高。

        \subsection{过渡晶体}

            纯粹的典型晶体不多，大多数晶体是它们之间的过渡晶体。

            \begin{itemize}
                \item \cemh{Na2O}、\cemh{MgO}：偏离子晶体，当作离子晶体
                \item \cemh{Al2O3}：看晶型
                \item \cemh{SiO2}：偏共价晶体，当作共价晶体
                \item \cemh{P2O5}、\cemh{SO3}、\cemh{Cl2O7}：分子晶体
                \item \cemh{TiCl4}：以共价键为主的过渡型晶体，偏分子晶体（熔点\SI{-25}{\celsius}）
                \item \cemh{TiO2}：共价键为主的过渡型晶体，偏共价晶体（熔点\SI{1830}{\celsius}）
            \end{itemize}

            \textcolor{blue}{离子液体中作用力有氢键、范德华力，处在离子晶体向分子晶体的过渡之中。}

    \section{晶体密度计算}\label{sect:晶体密度计算}

        \textcolor{red}{根据晶胞结构确定各种微粒的数目 \(\to\) 晶胞质量 \(\to\) 晶体密度 \(=\) 晶胞质量 \(\div\) 晶胞体积。}

        \begin{xmp}[密度计算]{典型晶体密度计算}
            \textbf{氯化铯}：已知最近两个 \cemh{Cs+} 核间距\SI{4.1E-10}{\m}。
            相对原子质量：\cemh{Cs} 133，\cemh{Cl} 35.5。
            \(\rho = \dfrac{1\times M(\cemh{CsCl})}{N_\text{A} \times a^3}\)

            \textbf{氯化钠}：已知 \cemh{Na+} 半径 \SI{101}{\pm}，\cemh{Cl-} 半径 \SI{181}{\pm}。
            \(a = 2(r(\cemh{Na+}) + r(\cemh{Cl-}))\)

            \textbf{金刚石}：已知晶胞边长 \SI{3.57E-10}{\m}。
            \(\rho = \dfrac{8 \times 12}{N_\text{A} \times a^3} = \SI{3.50}{\g\per\cubic\cm}\)

            \textbf{干冰}：已知晶胞边长 \SI{0.57}{\nm}。
            \(\rho = \dfrac{4 \times 44}{N_\text{A} \times a^3} = \SI{1.578}{\g\per\cubic\cm}\)
        \end{xmp}

    \section{晶体 X 射线衍射}\label{sect:晶体X射线衍射}

        在一定波长的 X 射线照射下，每种晶体都会显示出特有的衍射图像。

        可测定：
        \begin{itemize}
            \item 晶胞的大小和形状
            \item 晶胞中原子的种类、数目及位置排列
            \item 分子中化学键的键长、键角和空间结构
        \end{itemize}

        \textcolor{blue}{具有不损伤样品、测量精度高、信息量大等优点。}
