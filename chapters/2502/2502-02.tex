\section{\S 4.2 化学电源}\label{sect:化学电源}

    \subsection{普通锌锰干电池}\label{subsec:普通锌锰干电池}

        物质组成：

        \cemh{NH4Cl}、\cemh{MnO2}……

        \cemh{NH4+ <=> NH3 + H+}

        \begin{itemize}
            \item \cemh{2NH4+ + 2e- \xleq{} 2NH3 + H2 ^}
            \item \cemh{2MnO2 + H2 \xleq{} 2MnOOH}
        \end{itemize}

        \textcolor{blue}{\cemh{MnO2} 用于吸收 \cemh{H2} 防止极化。}

    \subsection{碱性锌锰干电池}\label{subsec:碱性锌锰干电池}

        \(+\) \cemh{KOH}

        \(\Rightarrow\) \cemh{Zn^{2+} + 2OH- \xleq{} Zn(OH)2 v}

        ——一次电池

        \textcolor{blue}{用 \cemh{KOH} 代替 \cemh{NH4Cl}，比能量更高，放电更稳定。}

    \subsection{二次电池}\label{subsec:二次电池}

        \subsubsection{铅蓄电池}\label{subsubsec:铅蓄电池}

            正：\cemh{PbO2 + 4H+ + SO4^{2-} + 2e- \xleq{} PbSO4 + 2H2O}

            充电（正正负负）

            负：\cemh{Pb + SO4^{2-} - 2e- \xleq{} PbSO4}（微溶）

            总：\cemh{Pb + PbO2 + 4H+ + 2SO4^{2-} \xleq{} 2PbSO4 + 2H2O}

            \textcolor{red}{可根据硫酸密度判断充电程度。}

        \subsubsection{\texorpdfstring{\cemh{Li+}}{Li+} 电池}\label{subsubsec:锂离子电池}

            \textcolor{blue}{锂电池：负极 \cemh{Li - e- \xleq{} Li+}（需用非水电解质）}

            \textcolor{blue}{锂离子电池：\cemh{Li+} 在正负极间「摇椅式」脱出和嵌入。}

            \begin{xmp}[锂离子电池]{\cemh{LiCoO2}—\cemh{C6} 锂电池}
                正：\(x\)\cemh{Li+ + Li_{1-x}CoO2 + $x$e- \xleq{} LiCoO2}

                负：\cemh{Li_{1-x}C6 - $x$e- \xleq{} C6 + $x$Li+}

                \textcolor{blue}{看 \cemh{Li+} 写电荷比标 \cemh{Co} 化合价容易。}

                负极材料：层状石墨材料

                正极材料：含锂的化合物
            \end{xmp}

    \subsection{氢氧燃料电池}\label{subsec:氢氧燃料电池}

        总反应：\cemh{2H2 + O2 \xleq{} 2H2O}

        \begin{center}
        \setlength{\tabcolsep}{3pt}
        \resizebox{\textwidth}{!}{\begin{tabular}{|c|c|c|c|}
            \hline
            \textbf{介质} & \textbf{负极反应} & \textbf{正极反应} & \textbf{极区 pH} \\
            \hline
            酸性 & \makecell[l]{\cemh{2H2 - 4e- -> 4H+}} & \makecell[l]{\cemh{O2 + 4H+ + 4e- -> 2H2O}} & 负极\(\downarrow\)\par 正极\(\uparrow\) \\
            \hline
            碱性 & \makecell[l]{\cemh{2H2 + 4OH- - 4e- -> 4H2O}} & \makecell[l]{\cemh{O2 + 2H2O + 4e- -> 4OH-}} & 负极\(\downarrow\)\par 正极\(\uparrow\) \\
            \hline
            熔融氧化物 & \makecell[l]{\cemh{2H2 + 2O^{2-} - 4e- -> 2H2O}} & \makecell[l]{\cemh{O2 + 4e- -> 2O^{2-}}} & --- \\
            \hline
            碳酸盐 & \makecell[l]{\cemh{2H2 + 2CO3^{2-} - 4e- -> 2H2O + 2CO2}} & \makecell[l]{\cemh{O2 + 2CO2 + 4e- -> 2CO3^{2-}}} & --- \\
            \hline
        \end{tabular}}
        \end{center}

        \textcolor{red}{极区 pH 变化规律：主要看电极反应，不是看总反应。}

        \textcolor{blue}{正正负负}

        \textcolor{blue}{碱性溶液不容易腐蚀外壳金属。}

        \begin{xmp}[甲烷燃料电池]{\cemh{CH4} 在碱性环境中的负极反应}
            负极：\cemh{CH4 + 10OH- - 8e- \xleq{} CO3^{2-} + 7H2O}

            总反应：\cemh{CH4 + 2O2 + 2OH- \xleq{} CO3^{2-} + 3H2O}
        \end{xmp}

        \begin{xmp}[甲醇燃料电池]{\cemh{CH3OH} 燃料电池}
            已知在有水条件下，液态 \cemh{CH3OH} 在负极上被氧化，产生 \cemh{CO2} 和 \cemh{H+} 并流出电子；\cemh{H+} 经过质子交换膜，在正极上与 \cemh{O2} 和流入的电子反应生成 \cemh{H2O}。

            负极：\cemh{CH3OH + H2O - 6e- \xleq{} CO2 ^ + 6H+}

            正极：\cemh{O2 + 4H+ + 4e- \xleq{} 2H2O}

            总反应：\cemh{2CH3OH + 3O2 \xleq{} 2CO2 + 4H2O}

            \textcolor{blue}{电解质溶液换成 \cemh{KOH} 溶液：}

            负极：\cemh{CH3OH + 8OH- - 6e- \xleq{} CO3^{2-} + 6H2O}
        \end{xmp}
